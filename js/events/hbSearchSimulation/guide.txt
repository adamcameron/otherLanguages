page load
	-> load settings
	-> get properties
		-> change page(1)
			-> activate controls (stops listening for change page)
												
															

change sort -> change setting (sort) -> sort -> change page(1)
										
change filter -> change settings (filters) -> filter -> sort

change page -> change page setting (page=n)	-> render page
											-> render pagination

resize page -> change page setting (size) -> change page(1)

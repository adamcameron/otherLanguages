	-> deactivate controls (off in the mark-up) .on("start") .trigger("next")
	-> load settings .on("start") .trigger("next")
	-> get properties			 .on("start")	.trigger("next")	
		-> change page(1) .on("next next next")
			-> activate controls (stops listening for change page)

change filter -> change settings (filters) -> filter -> sort

											-> render pagination
